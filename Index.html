  <script>

    // Load Google Charts packages

    google.charts.load('current', { packages: ['corechart', 'table'] });

    google.charts.setOnLoadCallback(init);


    // current filter state

    let currentFilters = { year: "ALL", pubType: "ALL", oa: "ALL", lang: "ALL" };


    // Keep latest summary to redraw on resize (so charts don't look squished)

    let lastSummary = null;


    function init() {

      google.script.run.withSuccessHandler(fillFilters).getFilterOptions();

      refreshAll();


      // redraw charts when window size changes (keeps it responsive)

      window.addEventListener("resize", () => {

        if(lastSummary) drawCharts(lastSummary);

      });

    }


    // ---------- Filters ----------

    function fillFilters(opt) {

      setSelect("fYear", opt.years);

      setSelect("fPub", opt.pubTypes);

      setSelect("fOA", opt.openAccess);

      setSelect("fLang", opt.languages);

    }


    function setSelect(id, arr) {

      const el = document.getElementById(id);

      const options = [`<option value="ALL">ALL</option>`]

        .concat((arr || []).map(v => `<option value="${escapeHtml(v)}">${escapeHtml(v)}</option>`))

        .join("");

      el.innerHTML = options;

    }


    function applyFilters() {

      currentFilters = {

        year: document.getElementById("fYear").value,

        pubType: document.getElementById("fPub").value,

        oa: document.getElementById("fOA").value,

        lang: document.getElementById("fLang").value

      };

      refreshAll();

    }


    function resetFilters() {

      document.getElementById("fYear").value = "ALL";

      document.getElementById("fPub").value = "ALL";

      document.getElementById("fOA").value = "ALL";

      document.getElementById("fLang").value = "ALL";

      applyFilters();

    }


    // ---------- Refresh Dashboard ----------

    function refreshAll() {

      refreshTable();

      refreshCharts();

    }


    function refreshTable() {

      google.script.run

        .withSuccessHandler(drawTable)

        .withFailureHandler(showError)

        .getRawDataFiltered(currentFilters, 10);

    }


    function refreshCharts() {

      google.script.run

        .withSuccessHandler((summary) => {

          lastSummary = summary;

          drawCharts(summary);

        })

        .withFailureHandler(showError)

        .getChartSummary(currentFilters);

    }


    // ---------- Table ----------

    function drawTable(payload) {

      const headers = payload.headers || [];

      const rows = payload.rows || [];

      const total = payload.total || 0;


      document.getElementById("resultCount").textContent =

        `Showing ${rows.length} rows (preview) out of ${total} matched rows.`;


      const dt = new google.visualization.DataTable();


      // make all columns string for safety

      headers.forEach(h => dt.addColumn("string", String(h)));


      const safeRows = rows.map(r =>

        (r || []).map(v => (v === null || v === undefined) ? "" : String(v))

      );


      dt.addRows(safeRows);


      const table = new google.visualization.Table(document.getElementById("dataTable"));

      table.draw(dt, {

        showRowNumber: true,

        width: "100%",

        height: "420px",

        allowHtml: true

      });

    }


    // ---------- Charts ----------

    function toArray(obj, col1Name) {

      const arr = [[col1Name, "Count"]];

      Object.keys(obj || {}).sort().forEach(k => {

        arr.push([String(k), Number(obj[k])]);

      });

      return arr;

    }


    function drawCharts(summary) {

      const base = {

        backgroundColor: "transparent",

        chartArea: { left: 52, top: 30, right: 18, bottom: 48, width: "100%", height: "100%" },

        legend: { textStyle: { fontName: "Inter", fontSize: 12 } },

        titleTextStyle: { fontName: "Inter", fontSize: 14, bold: true },

        tooltip: { textStyle: { fontName: "Inter", fontSize: 12 } }

      };


      // Year (Column chart)

      const yearDT = google.visualization.arrayToDataTable(toArray(summary.byYear, "Year"));

      new google.visualization.ColumnChart(document.getElementById("chartYear"))

        .draw(yearDT, {

          ...base,

          title: "Publications by Year",

          legend: "none",

          hAxis: { textStyle: { fontName: "Inter", fontSize: 12 } },

          vAxis: { textStyle: { fontName: "Inter", fontSize: 12 }, minValue: 0 }

        });


      // Publication type (Donut)

      const pubDT = google.visualization.arrayToDataTable(toArray(summary.byPub, "Publication type"));

      new google.visualization.PieChart(document.getElementById("chartPub"))

        .draw(pubDT, {

          ...base,

          title: "Publication Type",

          pieHole: 0.55,

          legend: { position: "right", textStyle: { fontName: "Inter", fontSize: 12 } }

        });


      // Open Access (Donut)

      const oaDT = google.visualization.arrayToDataTable(toArray(summary.byOA, "Open Access"));

      new google.visualization.PieChart(document.getElementById("chartOA"))

        .draw(oaDT, {

          ...base,

          title: "Open Access",

          pieHole: 0.55,

          legend: { position: "right", textStyle: { fontName: "Inter", fontSize: 12 } }

        });


      // Language (Donut)

      const langDT = google.visualization.arrayToDataTable(toArray(summary.byLang, "Language"));

      new google.visualization.PieChart(document.getElementById("chartLang"))

        .draw(langDT, {

          ...base,

          title: "Language",

          pieHole: 0.55,

          legend: { position: "right", textStyle: { fontName: "Inter", fontSize: 12 } }

        });

    }


    // ---------- Download ----------

    function downloadCsv() {

      google.script.run

        .withSuccessHandler((url) => window.open(url, "_blank"))

        .withFailureHandler(showError)

        .exportCsv(currentFilters);

    }


    // ---------- Helpers ----------

    function escapeHtml(s) {

      return String(s).replace(/[&<>"']/g, m => ({

        "&": "&amp;",

        "<": "&lt;",

        ">": "&gt;",

        '"': "&quot;",

        "'": "&#39;"

      }[m]));

    }


    function showError(err) {

      console.error(err);

      alert("Error: " + (err && err.message ? err.message : err));

    }
