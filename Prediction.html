  <script>

    //  Load ASJC list on page load

    window.addEventListener("load", () => {

      google.script.run

        .withSuccessHandler(fillAsjcDropdown)

        .withFailureHandler(showError)

        .getAsjcFieldOptions();

    });


    function fillAsjcDropdown(fields){

      const el = document.getElementById("asjc_field");

      const arr = (fields && fields.length) ? fields : ["All"];

      el.innerHTML = arr.map(v => `<option value="${escapeHtml(v)}">${escapeHtml(v)}</option>`).join("");


      // default selection

      el.value = arr.includes("Computer Science") ? "Computer Science" : arr[0];


      // toggle features when "All" selected

      el.addEventListener("change", toggleFeaturesForAll);

      toggleFeaturesForAll();

    }


    function toggleFeaturesForAll(){

      const field = document.getElementById("asjc_field").value;

      const isAll = String(field).toLowerCase() === "all";


      // If "All", features are not really meaningful (national sum uses model defaults).

      // We disable them to avoid confusion.

      document.querySelectorAll(".feat").forEach(inp => {

        inp.disabled = isAll;

        if(isAll) inp.classList.add("disabled");

        else inp.classList.remove("disabled");

      });


      const msg = document.getElementById("msg");

      if(isAll){

        msg.className = "msg";

        msg.innerHTML = '<span class="pill">All selected: will predict national total (sum of all fields).</span>';

      } else {

        msg.textContent = "";

        msg.className = "msg";

      }

    }


    function payloadFromUI(){

      return {

        asjc_field: document.getElementById("asjc_field").value.trim(),

        year: Number(document.getElementById("year").value),

        citations: Number(document.getElementById("citations").value),

        fwci: Number(document.getElementById("fwci").value),

        authors: Number(document.getElementById("authors").value),

        institutions: Number(document.getElementById("institutions").value),

        countries: Number(document.getElementById("countries").value)

      };

    }


    function setLoading(isLoading, text){

      const btn = document.getElementById("btnPredict");

      btn.disabled = isLoading;


      const msg = document.getElementById("msg");

      msg.className = "msg";


      if(!text){

        // keep “All” hint if All is selected

        toggleFeaturesForAll();

        return;

      }


      msg.innerHTML = '<span class="pill">' + text + '</span>';

    }


    function clearResult(){

      document.getElementById("kpis").style.display = "none";

      toggleFeaturesForAll();

    }


    function fillSample(){

      // if dropdown doesn't include Computer Science, use current selection

      const asjc = document.getElementById("asjc_field");

      if ([...asjc.options].some(o => o.value === "Computer Science")) {

        asjc.value = "Computer Science";

      }

      document.getElementById("year").value = "2027";

      document.getElementById("citations").value = "120";

      document.getElementById("fwci").value = "1.25";

      document.getElementById("authors").value = "5";

      document.getElementById("institutions").value = "3";

      document.getElementById("countries").value = "2";

      clearResult();

    }


    function predictNow(){

      clearResult();


      const p = payloadFromUI();

      if(!p.asjc_field){

        const msg = document.getElementById("msg");

        msg.className = "msg err";

        msg.textContent = "Please select ASJC Field.";

        return;

      }


      setLoading(true, "Predicting… calling Python API");


      google.script.run

        .withSuccessHandler(showPrediction)

        .withFailureHandler(showError)

        .getRealtimePrediction(p);

    }


    function showPrediction(res){

      setLoading(false, "Done ✅");


      const ets = (res && res.ETS_Pred != null) ? Number(res.ETS_Pred) : null;

      const r   = (res && res.Residual_Pred != null) ? Number(res.Residual_Pred) : null;

      const h   = (res && res.Hybrid_Pred != null) ? Number(res.Hybrid_Pred) : null;


      if(ets === null && r === null && h === null){

        const msg = document.getElementById("msg");

        msg.className = "msg err";

        msg.textContent =

          "API response did not include ETS_Pred / Residual_Pred / Hybrid_Pred. Check Python /predict output.";

        return;

      }


      document.getElementById("k_ets").textContent = (ets === null ? "-" : ets.toFixed(2));

      document.getElementById("k_res").textContent = (r === null ? "-" : r.toFixed(2));

      document.getElementById("k_hyb").textContent = (h === null ? "-" : h.toFixed(2));

      document.getElementById("kpis").style.display = "grid";

    }


    function showError(err){

      setLoading(false, "");

      const msg = document.getElementById("msg");

      msg.className = "msg err";

      msg.textContent = "Error: " + (err && err.message ? err.message : err);

    }


    function escapeHtml(s) {

      return String(s).replace(/[&<>"']/g, m => ({

        "&": "&amp;",

        "<": "&lt;",

        ">": "&gt;",

        '"': "&quot;",

        "'": "&#39;"

      }[m]));

    }
